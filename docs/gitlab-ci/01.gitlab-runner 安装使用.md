## 1. 安装
### 1.1 容器安装
```
docker run -d --name gitlab-runner --restart always \
-v /srv/gitlab-runner/config:/etc/gitlab-runner \
-v /var/run/docker.sock:/var/run/docker.sock \
gitlab/gitlab-runner:latest
```

### 1.2 普通安装
```
sudo curl -L --output /usr/local/bin/gitlab-runner https://gitlab-runner-downloads.s3.amazonaws.com/latest/binaries/gitlab-runner-linux-amd64

sudo chmod +x /usr/local/bin/gitlab-runner

curl -sSL https://get.docker.com/ | sh

sudo useradd --comment 'GitLab Runner' --create-home gitlab-runner --shell /bin/bash

sudo gitlab-runner install --user=gitlab-runner --working-directory=/home/gitlab-runner
sudo gitlab-runner start
```

## 3. .gitlab-ci.yml文件
```
# 在作业之前执行的脚本或命令
before_script:

# 在作业之后执行的脚本或命令
after_script:

# 依赖的docker镜像
#image:

# 依赖的docker服务
#services:

# 定义工作场景，和场景顺序
stages:
  - build
  - push
  - deploy

# 定义任务，任务名字随意
build_image_dev:
  # 定义所属场景
  stage: build
  # 定义任务脚本
  script:
    - docker build -t register.ptl-harbor.com/test-dev:0.0.1 .
  # 定义分支
  only:
    - dev
  # 指定runner，tags为runner标签
  #tags:
  #  - dev

build_image_test:
  stage: build
  script:
    - docker build -t register.ptl-harbor.com/test-test:0.0.1 .
  only:
    - test

build_image_prd:
  stage: build
  script:
    - docker build -t register.ptl-harbor.com/test-prd:0.0.1 .
  only:
    - master

push_images:
  stage: push
  script:
    - docker push register.ptl-harbor.com/test-dev:0.0.1
    - docker push register.ptl-harbor.com/test-test:0.0.1
    - docker push register.ptl-harbor.com/test-prd:0.0.1
```
